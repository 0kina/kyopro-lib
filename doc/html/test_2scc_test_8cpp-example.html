#include <iostream>
#include "scc.hpp"
using namespace std;
/*
* (0) --> (1) (8) --> (9)
* ^ | ^
* | v |
* (3) <-- (2) --> (6) --> (5)
* ^ |
* | v
* (4) <-- (7)
*/
int main() {
int n = 10;
vector<vector<int>> graph(n);
graph[0].push_back(1);
graph[1].push_back(2);
graph[2].push_back(3);
graph[2].push_back(6);
graph[3].push_back(0);
graph[4].push_back(6);
graph[5].push_back(7);
graph[5].push_back(8);
graph[6].push_back(5);
graph[7].push_back(4);
graph[8].push_back(9);
graph[9].push_back(9);
SCC scc(graph);
vector<int> comp_ids = scc.comp();
for (int i = 0; i < n; ++i) {
cout << i << "th vertex: " << comp_ids[i] << "th component, whose size is ";
cout << scc.comp_size(i) << endl;
}
vector<vector<int>> components = scc.scc();
for (int i = 0; i < components.size(); ++i) {
cout << i << "th component:\n";
for (int v : components[i]) {
cout << v << " ";
}
cout << endl;
}
}
