* Restored: 4X^3 + 3X^2 + 2X^1 + 1X^0
* F: 3X^3 + 7X^2 + 8X^1 + 1X^0
* G: 1X^2 + 3X^1 + 2X^0
* Correct FG : 3X^5 + 16X^4 + 35X^3 + 39X^2 + 19X^1 + 2X^0
* Calculated FG: 3X^5 + 16X^4 + 35X^3 + 39X^2 + 19X^1 + 2X^0
*/
template <typename T>
std::string func_to_string(vector<T> f, int n) {
ostringstream ost;
for (int i = n - 1; i >= 0; --i) {
ost << f[i] << "X^" << i;
if (i > 0) ost << " + ";
}
return ost.str();
}
